```tsx
"use client";

import { useEffect, useState } from "react";
import { useRouter } from "next/navigation";

export default function SwipePage() {
  const router = useRouter();
  const [username, setUsername] = useState<string | null>(null);
  // null = "not checked yet" (pre-hydration)
  // non-null string = "checked and found"
  // After check: if no username → redirect; if found → render

  useEffect(() => {
    const stored = sessionStorage.getItem("username");
    if (!stored) {
      router.replace("/login");
    } else {
      setUsername(stored);
    }
  }, [router]);

  // Return null while checking sessionStorage — avoids flash of unprotected content.
  // The check is <1ms (synchronous localStorage read); no spinner needed.
  if (username === null) return null;

  return <div>Swipe page — Phase 3</div>;
}
```

Key constraints:
- `useState<string | null>(null)` — null means "not yet checked", prevents FOUC
- `return null` when username === null — renders nothing until guard fires; avoids flash
- `router.replace("/login")` — replace not push, prevents back-navigation to unguarded state
- `sessionStorage.getItem("username")` is called INSIDE useEffect, never at top level (SSR safety)
- Placeholder body `<div>Swipe page — Phase 3</div>` — Phase 3 replaces only this div, guard stays

| sessionStorage → React state | Stored username read back on /swipe and /page and rendered to DOM |
| sessionStorage → router.replace URL | Hard-coded redirect targets ("/swipe", "/login") — not user-controlled |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04 | Tampering | Username input — empty/whitespace submission | mitigate | `username.trim().length > 0` guard in handleSubmit before setItem; empty string early-returns |
| T-02-05 | Spoofing | Username displayed in UI (Phase 3 Navbar) | accept (mitigated by React) | React JSX auto-escapes all text content rendered via `{username}`; no `dangerouslySetInnerHTML` used |
| T-02-06 | Tampering | sessionStorage poisoning via DevTools | accept | Username is display-only; no privilege, no authorization check, no secret; tampering has no security consequence |
| T-02-07 | Elevation of Privilege | Bypassing auth guard by manually navigating to /swipe | accept | This is not real authentication per spec — username is for record-keeping only; no sensitive data behind guard |
| T-02-08 | Spoofing | XSS via crafted username value | accept | React escapes all JSX text interpolation; no innerHTML assignment in this phase |
</threat_model>
